```javascript JavaScript
// Place a market order
const order = await fetch('https://api.hypiq.finance/v1/orders', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    symbol: 'BTCUSD',
    side: 'buy',
    type: 'market',
    quantity: '0.001'
  })
});
```
